Python
import requests

token = request.form.get('_antispam_token', '')
response = requests.post('{{ verify_url }}', json={
    'secret': '{{ project['private_key'] }}',
    'token': token
})
result = response.json()

if result.get('success') and result.get('score', 0) >= 0.5:
    # Human
    pass
else:
    # Bot
    pass
